Add after the issue change section, before the issues section.

```python
# --- Ingest ---


def _next_ingest_id() -> str:
    with _get_conn() as conn:
        row = conn.execute("SELECT COUNT(*) as c FROM records WHERE type = 'ingest'").fetchone()
        num = row["c"] + 1 
        return f"ingest_{num:03d}"


def save_ingest(
    source: str,
    raw: str,
    human_context: str = "",
) -> str:
    """Save a seed ingest record to the SOR.

    Auto-classified as 'relevant' — bypasses triage.
    Returns record_id (e.g. 'ingest_001').
    """
    init_db()
    with _get_conn() as conn:
        record_id = _next_ingest_id()
        now = _now()

        conn.execute(
            "INSERT INTO records (id, type, source, created_at, raw, classification, human_context) "
            "VALUES (%s, %s, %s, %s, %s, %s, %s)",
            (record_id, "ingest", source, now, raw, "relevant", human_context or None),
        )
        conn.commit()
        return record_id
```

No new table — raw seed content goes into the `raw` column of `records`.

def execute(self, **kwargs: Any) -> str:
    import os
    from pearscarf import store

    file_path = kwargs["file_path"]
    if not os.path.isfile(file_path):
        return f"Error: file not found: {file_path}"

    with open(file_path) as fh:
        raw = fh.read()

    if not raw.strip():
        return f"Error: file is empty: {file_path}"

    record_id = store.save_ingest(
        source="ingest_expert",
        raw=raw,
    )

### `ParseRecordFileTool.execute()`

```python
def execute(self, **kwargs: Any) -> str:
    import json
    import os
    from pearscarf import store

    file_path = kwargs["file_path"]
    record_type = kwargs["record_type"]

    if not os.path.isfile(file_path):
        return f"Error: file not found: {file_path}"

    with open(file_path) as fh:
        data = json.load(fh)

    # Accept a single record dict or a list
    records = data if isinstance(data, list) else [data]

    saved = 0
    skipped = 0

    for rec in records:
        if record_type == "email":
            result = store.save_email(
                source="ingest_expert",
                sender=rec.get("sender", ""),
                subject=rec.get("subject", ""),
                body=rec.get("body", ""),
                message_id=rec.get("message_id"),
                recipient=rec.get("recipient", ""),
                received_at=rec.get("received_at", ""),
                raw=json.dumps(rec),
            )
            if result is None:
                skipped += 1
            else:
                # Auto-classify as relevant
                store.classify_record(result, "relevant", reason="ingested via file")
                saved += 1

        elif record_type == "issue":
            record_id, is_new = store.save_issue(
                source="ingest_expert",
                linear_id=rec.get("linear_id", rec.get("id", "")),
                identifier=rec.get("identifier", ""),
                title=rec.get("title", ""),
                description=rec.get("description", ""),
                status=rec.get("status", ""),
                priority=rec.get("priority", ""),
                assignee=rec.get("assignee", ""),
                project=rec.get("project", ""),
                labels=rec.get("labels"),
                comments=rec.get("comments"),
                url=rec.get("url", ""),
                linear_created_at=rec.get("linear_created_at", ""),
                linear_updated_at=rec.get("linear_updated_at", ""),
                raw=json.dumps(rec),
            )
            if is_new:
                store.classify_record(record_id, "relevant", reason="ingested via file")
                saved += 1
            else:
                skipped += 1

        elif record_type == "issue_change":
            result = store.save_issue_change(
                issue_record_id=rec.get("issue_record_id", ""),
                field=rec.get("field", ""),
                from_value=rec.get("from_value", ""),
                to_value=rec.get("to_value", ""),
                linear_history_id=rec.get("linear_history_id"),
                changed_by=rec.get("changed_by", ""),
                changed_at=rec.get("changed_at", ""),
            )
            if result is None:
                skipped += 1
            else:
                saved += 1  # already auto-classified by save_issue_change

    parts = [f"Saved {saved} {record_type} record(s)."]
    if skipped:
        parts.append(f"Skipped {skipped} duplicate(s).")
    return " ".join(parts)
```

## 4. `pearscarf/prompts/ingest_extraction.md`

Dedicated extraction prompt for seed ingest records. The input is a typed block markdown file with sections like `## people`, `## companies`, `## projects`, `## facts`. The output is the same JSON schema as the standard extraction prompt.

```markdown
You are an entity extraction system for seed data files. Seed files contain structured information about people, companies, projects, and their relationships. All content is declared ground truth.

## Input Format

The input is a markdown file with typed sections:

- `## people` — one person per line or block, with name, email, role, company
- `## companies` — one company per line or block, with name, domain
- `## projects` — one project per line or block, with name and description
- `## facts` — one fact per line, with category, from_entity, to_entity, and fact text

## Entity Name Normalization

Use the simplest, most common name for every entity. Same rules as standard extraction:

- **Companies**: Strip legal suffixes. "Acme" not "Acme, Inc."
- **People**: Full formal name. "Michael Chen" not "Mike" or "michael@acme.com"
- **Projects**: Short working name. "Series A" not "Series A Financing Round"

Do not include domains, emails, or parenthetical qualifiers in entity names — those go in metadata.

## Extraction Rules

- Every person entry → `person` entity with `email` and `role` in metadata (when stated)
- Every company entry → `company` entity with `domain` in metadata (when stated)
- Every project entry → `project` entity
- Every fact entry → a fact edge with the declared category
- **confidence**: always `stated` — seed data is declared ground truth, never inferred
- **valid_at**: use the date from the fact if one is provided, otherwise null

## Fact Categories

Structural: WORKS_AT, FOUNDED, MANAGES, PART_OF, MEMBER_OF
Activity: COMMUNICATED, STATUS_CHANGED, MENTIONED_IN
Claims: COMMITTED_TO, DECIDED, BLOCKED_BY, EVALUATED

## Output

Return a single JSON object with exactly two keys. No markdown fences, no preamble.

{
  "entities": [
    {
      "type": "person | company | project | event",
      "name": "Canonical Name",
      "metadata": { "email": "...", "role": "...", "domain": "..." }
    }
  ],
  "facts": [
    {
      "category": "WORKS_AT",
      "fact": "Michael Chen works at Acme as CTO",
      "from_entity": "Michael Chen",
      "to_entity": "Acme",
      "confidence": "stated",
      "valid_at": "2026-01-15"
    }
  ]
}
```
