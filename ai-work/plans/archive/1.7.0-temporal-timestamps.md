# Plan: 1.7.0 — Temporal Timestamps

Bi-temporal modeling for all graph edges. Every relationship and fact gets `valid_at` / `invalid_at` timestamps. Facts use invalidate-and-create instead of update-in-place. Retriever queries default to current state. Existing data retrofitted. Memory CLI shows temporal info.

**`create_edge`** — add `valid_at` and `invalid_at` to relationship properties:

```python
def create_edge(
    from_entity: str,
    to_entity: str,
    relationship: str,
    source_record: str,
    valid_at: str | None = None,
) -> str:
    rel_type = relationship.upper()
    ts = _now()
    props = {
        "source_record": source_record,
        "created_at": ts,
        "valid_at": valid_at or ts,
        "invalid_at": None,
    }

    with get_session() as session:
        result = session.run(
            "MATCH (a) WHERE elementId(a) = $from_id "
            "MATCH (b) WHERE elementId(b) = $to_id "
            "CALL apoc.create.relationship(a, $rel_type, $props, b) "
            "YIELD rel "
            "RETURN elementId(rel) AS rid",
            from_id=from_entity,
            to_id=to_entity,
            rel_type=rel_type,
            props=props,
        )
        record = result.single()
        return record["rid"] if record else ""
```

**`invalidate_edge`** — new function to close out a relationship:

```python
def invalidate_edge(edge_id: str, invalid_at: str | None = None) -> None:
    """Set invalid_at on an existing relationship."""
    ts = invalid_at or _now()
    with get_session() as session:
        session.run(
            "MATCH ()-[r]->() WHERE elementId(r) = $rid "
            "SET r.invalid_at = $ts",
            rid=edge_id,
            ts=ts,
        )
```

**`upsert_fact`** — replace MERGE with invalidate-and-create:

```python
def upsert_fact(
    entity_id: str,
    claim: str,
    confidence: str,
    source_record: str,
    valid_at: str | None = None,
) -> str:
    ts = _now()
    vat = valid_at or ts

    with get_session() as session:
        # Step 1: Find existing current fact with same claim on this entity
        existing = session.run(
            "MATCH (n)-[:HAS_FACT]->(f:Fact {claim: $claim}) "
            "WHERE elementId(n) = $eid AND f.invalid_at IS NULL "
            "RETURN elementId(f) AS fid",
            eid=entity_id,
            claim=claim,
        ).single()

        # Step 2: Invalidate old fact if it exists
        if existing:
            session.run(
                "MATCH (f:Fact) WHERE elementId(f) = $fid "
                "SET f.invalid_at = $ts",
                fid=existing["fid"],
                ts=vat,
            )

        # Step 3: Create new fact
        result = session.run(
            "MATCH (n) WHERE elementId(n) = $eid "
            "CREATE (n)-[:HAS_FACT]->(f:Fact {"
            "  claim: $claim, confidence: $confidence, "
            "  source_record: $source, created_at: $created, "
            "  valid_at: $valid_at, invalid_at: null"
            "}) "
            "RETURN elementId(f) AS fid",
            eid=entity_id,
            claim=claim,
            confidence=confidence,
            source=source_record,
            created=ts,
            valid_at=vat,
        )
        record = result.single()
        return record["fid"] if record else ""
```

**`get_entity_facts`** — add `current_only` parameter (default `True`), include temporal fields:

```python
def get_entity_facts(entity_id: str, current_only: bool = True) -> list[dict]:
    with get_session() as session:
        where = "WHERE elementId(n) = $eid"
        if current_only:
            where += " AND f.invalid_at IS NULL"
        result = session.run(
            f"MATCH (n)-[:HAS_FACT]->(f:Fact) {where} "
            "RETURN f, elementId(f) AS fid",
            eid=entity_id,
        )
        facts = []
        for record in result:
            fnode = record["f"]
            facts.append({
                "id": record["fid"],
                "claim": fnode.get("claim", ""),
                "confidence": fnode.get("confidence", ""),
                "source_record": fnode.get("source_record", ""),
                "created_at": fnode.get("created_at", ""),
                "valid_at": fnode.get("valid_at", ""),
                "invalid_at": fnode.get("invalid_at"),
            })
        return facts
```

**`retrofit_temporal`** — new one-time migration function:

```python
def retrofit_temporal() -> dict:
    """Add temporal timestamps to existing edges and facts that lack them.
    Sets valid_at = created_at, invalid_at = null. Returns counts."""
    with get_session() as session:
        # Retrofit relationships
        result = session.run(
            "MATCH ()-[r]->() "
            "WHERE r.valid_at IS NULL AND type(r) <> 'HAS_FACT' "
            "SET r.valid_at = coalesce(r.created_at, datetime().epochMillis), "
            "    r.invalid_at = null "
            "RETURN count(r) AS c"
        )
        edges = result.single()["c"]

        # Retrofit HAS_FACT edges (set on the Fact node, not the edge)
        result = session.run(
            "MATCH (n)-[:HAS_FACT]->(f:Fact) "
            "WHERE f.valid_at IS NULL "
            "SET f.valid_at = coalesce(f.created_at, datetime().epochMillis), "
            "    f.invalid_at = null "
            "RETURN count(f) AS c"
        )
        facts = result.single()["c"]

        return {"edges_retrofitted": edges, "facts_retrofitted": facts}
```

### 2. `pearscaff/indexer.py` — Pass `valid_at` for change records

No change to how the Indexer calls `create_edge` or `upsert_fact` for emails and issues — they get `valid_at = now()` by default.

For `issue_change` records, pass `valid_at` from the change's `changed_at` timestamp. This requires reading the `changed_at` field from the record metadata and passing it through.

```python
# In _process_record, after extraction, before graph writes:
# If this is an issue_change, get the changed_at for valid_at
valid_at = None
if record.get("type") == "issue_change":
    with _get_conn() as conn:
        row = conn.execute(
            "SELECT changed_at FROM issue_changes WHERE record_id = %s",
            (record_id,),
        ).fetchone()
        if row:
            valid_at = str(row["changed_at"])

# Then pass valid_at to upsert_fact:
graph.upsert_fact(eid, claim, confidence, record_id, valid_at=valid_at)

# And to create_edge:
graph.create_edge(from_id, to_id, rel_type, record_id, valid_at=valid_at)
```

**`FactsLookupTool`** — add optional `include_superseded` parameter:

```python
input_schema = {
    "type": "object",
    "properties": {
        "entity_id": {
            "type": "string",
            "description": "Entity ID",
        },
        "include_superseded": {
            "type": "boolean",
            "description": "Include superseded (historical) facts. Default false.",
        },
    },
    "required": ["entity_id"],
}

def execute(self, **kwargs):
    entity_id = kwargs["entity_id"]
    include_superseded = kwargs.get("include_superseded", False)
    facts = graph.get_entity_facts(entity_id, current_only=not include_superseded)
    if not facts:
        return "No facts found for this entity."
    lines = []
    for f in facts:
        conf = f" [{f['confidence']}]" if f.get("confidence") else ""
        src = f" (from: {f['source_record']})" if f.get("source_record") else ""
        temporal = ""
        if f.get("invalid_at"):
            temporal = f" [was: {f['valid_at']} → {f['invalid_at']}]"
        elif f.get("valid_at"):
            temporal = f" [since: {f['valid_at']}]"
        lines.append(f"- {f['claim']}{conf}{src}{temporal}")
    return "\n".join(lines)
```
